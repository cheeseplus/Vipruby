def get_tenant_uid

    url: "#{@base_url}/tenant",
    headers: {:X-SDS-AUTH-TOKEN' => @auth_token,
      accept: :json
    },
    verify_ssl: @verify_cert
    ))
end

def rest_post(payload, api_url)
  JSON.parse(RestClient::Request.execute(method: :post,
       url: api_url,
       verify_ssl: @verify_cert,
       payload: payload,
       headers: {
         :X-SDS-AUTH-TOKEN' => @auth_token,
         content_type: 'application/json',
         accept: :json
       })).to_json
end
