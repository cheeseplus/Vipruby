def generate_initiators_json
  initiator_json = []
  @initiators_port.each do |initiator|
    initiator_json <<
    {
      protocol: @protocol.upcase,
      initiator_port: initiator,
      initiator_node: @initiator_node
    }.to_json
  end
  initiator_json
end
